# Binary Tree Node
class TreeNode:
    def __init__(self, val):
        self.val = val
        self.left = None
        self.right = None


# Function to validate BST
def isValidBST(root):

    def check(node, low, high):
        if node is None:
            return True

        if not (low < node.val < high):
            return False

        return (check(node.left, low, node.val) and
                check(node.right, node.val, high))

    return check(root, float('-inf'), float('inf'))


# -------------------------
# Example 1 : Valid BST
#         8
#       /   \
#      3     10
#     / \      \
#    1   6      14
# -------------------------

root1 = TreeNode(8)
root1.left = TreeNode(3)
root1.right = TreeNode(10)

root1.left.left = TreeNode(1)
root1.left.right = TreeNode(6)

root1.right.right = TreeNode(14)

print("Tree 1 Valid BST?", isValidBST(root1))


# -------------------------
# Example 2 : Invalid BST
#         10
#       /    \
#      5      15
#            /  \
#           6   20
# -------------------------

root2 = TreeNode(10)
root2.left = TreeNode(5)
root2.right = TreeNode(15)

root2.right.left = TreeNode(6)
root2.right.right = TreeNode(20)

print("Tree 2 Valid BST?", isValidBST(root2))
